-- JOINS INNER, OUTER(LEFT RIGHT FULL) --
CREATE TABLE course(
rollno INT PRIMARY KEY,
subject VARCHAR(10)
);

SELECT * 
FROM student
INNER JOIN course ON student.rollno = course.rollno; -- syntax same as LEFT and RIGHT JOIN --

SELECT * 
FROM student as s -- alias is the alternate name given to a table --
INNER JOIN course as c ON s.rollno = c.rollno; 

SELECT * 
FROM student
LEFT JOIN course ON student.rollno = course.rollno 
UNION  -- FULL join's syntax does not exist in MySql, it exists in Oracle --
SELECT * 
FROM student
RIGHT JOIN course ON student.rollno = course.rollno; 

SELECT * 
FROM student
LEFT JOIN course ON student.rollno = course.rollno
WHERE course.rollno is null; -- LEFT EXCLUSIVE join --
-- = null will not work --

SELECT * 
FROM student
RIGHT JOIN course ON student.rollno = course.rollno
WHERE student.rollno is null; -- RIGHT EXCLUSIVE join --

SELECT *
FROM student
INNER JOIN course ON student.rollno = course.rollno
WHERE student.rollno IS NOT NULL AND course .rollno IS NOT NULL; -- only intersection part --

SELECT *
FROM student as a
JOIN student as b
WHERE a.rollno = b.rollno;
